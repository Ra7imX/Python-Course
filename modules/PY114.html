Méthodes principales des listes

Objectifs d'apprentissage

  1. Maîtriser les principales méthodes des listes
  2. Comprendre la différence entre les méthodes
  3. Apprendre à modifier efficacement les listes
  4. Appliquer les méthodes dans des cas pratiques

1. Méthodes d'ajout

append() - Ajouter un élément à la fin :

villes = ["Casablanca", "Rabat"]
villes.append("Fès")
print(f"Après append : {villes}")

insert() - Insérer à une position spécifique :

villes.insert(1, "Tanger") # Insérer à l'index 1
print(f"Après insert : {villes}")

extend() - Ajouter plusieurs éléments :

autres_villes = ["Meknès", "Oujda"]
villes.extend(autres_villes)
print(f"Après extend : {villes}")

2. Méthodes de suppression

remove() - Supprimer par valeur :

plats = ["Tajine", "Couscous", "Pastilla", "Tajine"]
plats.remove("Tajine") # Supprime la première occurrence
print(f"Après remove : {plats}")

pop() - Supprimer par index et récupérer :

dernier_plat = plats.pop() # Supprime et retourne le dernier élément
plat_specific = plats.pop(0) # Supprime et retourne l'élément à l'index 0
print(f"Éléments supprimés : {dernier_plat}, {plat_specific}")

clear() - Vider la liste :

plats.clear()
print(f"Après clear : {plats}")

3. Méthodes de recherche et tri

index() - Trouver la position :

destinations = ["Marrakech", "Agadir", "Fès", "Tanger", "Marrakech"]
position = destinations.index("Marrakech") # Première occurrence
print(f"Position de Marrakech : {position}")

count() - Compter les occurrences :

occurrences = destinations.count("Marrakech")
print(f"Nombre de Marrakech : {occurrences}")

sort() - Trier la liste :

destinations.sort()
print(f"Liste triée : {destinations}")

reverse() - Inverser l'ordre :

destinations.reverse()
print(f"Liste inversée : {destinations}")

4. Exemple pratique - Gestion d'un restaurant

class RestaurantMarocain:
    def __init__(self):
        self.menu = []
        self.commandes = []

    def ajouter_plat(self, plat, prix):
        """Ajoute un plat au menu"""
        self.menu.append({"plat": plat, "prix": prix})
        print(f"Plat ajouté : {plat} à {prix} DH")

    def supprimer_plat(self, plat):
        """Supprime un plat du menu"""
        for item in self.menu:
            if item["plat"] == plat:
                self.menu.remove(item)
                print(f"Plat supprimé : {plat}")
                return
        print(f"Plat non trouvé : {plat}")

    def prendre_commande(self, table, plats):
        """Enregistre une nouvelle commande"""
        commande = {
            "table": table,
            "plats": plats.copy(),
            "statut": "en attente"
        }
        self.commandes.append(commande)
        print(f"Commande enregistrée pour la table {table}")

    def afficher_menu(self):
        """Affiche le menu complet"""
        print("\n=== Menu du Restaurant ===")
        for item in sorted(self.menu, key=lambda x: x["prix"]):
            print(f"{item['plat']} : {item['prix']} DH")

    def afficher_commandes(self):
        """Affiche toutes les commandes"""
        print("\n=== Commandes en cours ===")
        for commande in self.commandes:
            print(f"Table {commande['table']} - {commande['statut']}")
            for plat in commande["plats"]:
                print(f" - {plat}")

# Test du système de restaurant
restaurant = RestaurantMarocain()

# Ajout de plats
restaurant.ajouter_plat("Tajine Poulet", 80)
restaurant.ajouter_plat("Couscous Royal", 90)
restaurant.ajouter_plat("Pastilla", 70)
restaurant.ajouter_plat("Harira", 30)

# Affichage du menu
restaurant.afficher_menu()

# Prise de commandes
restaurant.prendre_commande(1, ["Tajine Poulet", "Harira"])
restaurant.prendre_commande(2, ["Couscous Royal", "Pastilla"])

# Affichage des commandes
restaurant.afficher_commandes()

# Suppression d'un plat
restaurant.supprimer_plat("Harira")

# Réaffichage du menu
restaurant.afficher_menu()
