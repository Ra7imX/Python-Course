Parcourir une liste - indices et éléments

Objectifs d'apprentissage

  1. Comprendre les différentes façons de parcourir une liste
  2. Maîtriser l'utilisation des indices
  3. Utiliser enumerate() efficacement
  4. Appliquer les techniques d'itération dans des cas pratiques

1. Parcours simple

villes_maroc = ["Casablanca", "Rabat", "Fès", "Marrakech"]

# Parcours des éléments
print("Parcours simple des villes:")
for ville in villes_maroc:
    print(f"Visite de {ville}")

# Parcours avec index manuel
print("\nParcours avec index:")
for i in range(len(villes_maroc)):
    print(f"Ville {i+1}: {villes_maroc[i]}")

2. Utilisation de enumerate()

sites_touristiques = ["Jamaa El Fna", "Médina de Fès", "Hassan II"]

print("\nParcours avec enumerate:")
for index, site in enumerate(sites_touristiques, 1):
    print(f"Site {index}: {site}")

3. Parcours de plusieurs listes simultanément

villes = ["Marrakech", "Agadir", "Tanger"]
hotels = ["Riad Luxe", "Palm Resort", "Marina Spa"]
prix = [1200, 800, 950]

print("\nInformations des hôtels:")
for ville, hotel, tarif in zip(villes, hotels, prix):
    print(f"{hotel} à {ville} - {tarif} DH/nuit")

4. Exemple pratique - Système de gestion de voyage

class GestionVoyage:
    def __init__(self):
        self.destinations = [
            {
                "ville": "Marrakech",
                "sites": ["Jamaa El Fna", "Jardins Majorelle", "Koutoubia"],
                "hotels": ["Riad Luxe", "Palais Oriental", "Atlas Resort"],
                "prix_hotels": [1500, 1200, 800]
            },
            {
                "ville": "Fès",
                "sites": ["Médina", "Bab Boujloud", "Médersa"],
                "hotels": ["Riad Fès", "Palais Médina", "Dar Traditional"],
                "prix_hotels": [1300, 1000, 700]
            }
        ]

    def afficher_details_destination(self, ville):
        for dest in self.destinations:
            if dest["ville"].lower() == ville.lower():
                print(f"\n=== Détails pour {dest['ville']} ===")
                print("\nSites touristiques:")
                for i, site in enumerate(dest['sites'], 1):
                    print(f"{i}. {site}")
                print("\nHôtels disponibles:")
                for hotel, prix in zip(dest['hotels'], dest['prix_hotels']):
                    print(f"- {hotel}: {prix} DH/nuit")
                return
        print(f"Destination {ville} non trouvée")

    def rechercher_hotels_budget(self, budget):
        print(f"\nHôtels disponibles pour un budget de {budget} DH/nuit:")
        hotels_trouves = False
        for dest in self.destinations:
            print(f"\nÀ {dest['ville']}:")
            for hotel, prix in zip(dest['hotels'], dest['prix_hotels']):
                if prix <= budget:
                    print(f"- {hotel} ({prix} DH)")
                    hotels_trouves = True
        if not hotels_trouves:
            print("Aucun hôtel trouvé dans ce budget")

    def planifier_circuit(self, duree_max):
        print(f"\nCircuit proposé ({duree_max} jours):")
        jours_restants = duree_max
        for i, dest in enumerate(self.destinations, 1):
            if jours_restants <= 0:
                break
            jours_ville = min(3, jours_restants) # Maximum 3 jours par ville
            print(f"\nÉtape {i}: {dest['ville']} ({jours_ville} jours)")
            print("Programme suggéré:")
            for jour in range(1, jours_ville + 1):
                sites_jour = dest['sites'][(jour-1)%len(dest['sites'])]
                print(f"Jour {jour}: Visite de {sites_jour}")
            jours_restants -= jours_ville

# Test du système
voyage = GestionVoyage()
voyage.afficher_details_destination("Marrakech")
voyage.rechercher_hotels_budget(1000)
voyage.planifier_circuit(5)

5. Notes techniques importantes

  • Méthodes de parcours :
    • Boucle for simple : pour les éléments
    • range + index : pour accès par position
    • enumerate : pour élément et position
    • zip : pour plusieurs listes en parallèle
  • Bonnes pratiques :
    • Choisir la méthode adaptée au besoin
    • Utiliser enumerate plutôt que range quand possible
    • Préférer les parcours simples sauf nécessité
  • Points d'attention :
    • Ne pas modifier la liste pendant l'itération
    • Vérifier les indices lors de l'accès direct
    • Gestion des listes de tailles différentes avec zip
